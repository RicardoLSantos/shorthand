#### Task 4.2.1: Fix Extension Context from #Element to Specific Types (2.5 hours)
**Impact:** -27 warnings

**Pattern to apply:** Change `* ^context[0].type = #Element` to specific resource type

**Example fix for mindfulness extensions:**
```fsh
// BEFORE:
Extension: MindfulnessContext
* ^context[0].type = #Element

// AFTER:
Extension: MindfulnessContext
* ^context[0].type = #Observation
* ^context[0].expression = "Observation"
// OR for multiple contexts:
* ^context[+].type = #QuestionnaireResponse
* ^context[=].expression = "QuestionnaireResponse"
```

**Automation script:**
```bash
# Create fix script
cat > fix_extension_contexts.sh <<'EOF'
#!/bin/bash
# Map extensions to appropriate context types
declare -A contexts=(
  ["activity-quality"]="#Observation"
  ["advanced-vital-signs-context"]="#Observation"
  ["environmental-context"]="#Observation"
  ["measurement-context"]="#Observation"
  ["mindfulness-context"]="#Observation"
  ["mindfulness-session-duration"]="#Observation"
  ["social-activity"]="#Observation"
  ["social-context"]="#Observation"
  ["social-support"]="#Observation"
  ["stress-coping"]="#Observation"
  ["stress-triggers"]="#Observation"
  ["audit-format"]="#AuditEvent"
  ["audit-level"]="#AuditEvent"
  ["audit-retention"]="#AuditEvent"
  ["mindfulness-audit-format"]="#AuditEvent"
  ["mindfulness-audit-level"]="#AuditEvent"
  ["mindfulness-audit-retention"]="#AuditEvent"
  ["regulatory-basis"]="#Consent"
  ["mindfulness-notification-enabled"]="#Patient"
  ["mindfulness-preferred-guide"]="#Patient"
  # Add others as needed
)

for ext in "${!contexts[@]}"; do
  file="input/fsh/extensions/${ext}.fsh"
  if [ -f "$file" ]; then
    sed -i.bak "s/\* \^context\[0\]\.type = #Element/\* \^context[0].type = ${contexts[$ext]}\n\* \^context[0].expression = \"${contexts[$ext]#\#}\"/" "$file"
    echo "Fixed: $file"
  fi
done
EOF
chmod +x fix_extension_contexts.sh
```
